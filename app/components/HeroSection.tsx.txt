'use client'

import { useEffect, useRef } from 'react'
import { ArrowRight, Sparkles, Shield, Zap } from 'lucide-react'

interface HeroContent {
  heading?: string
  subheading?: string
  cta_text?: string
  cta_link?: string
}

export default function HeroSection({ content, settings }: { content: HeroContent, settings: any }) {
  const canvasRef = useRef<HTMLCanvasElement>(null)
  
  const heading = content?.heading || 'We Build Amazing Websites'
  const subheading = content?.subheading || 'Professional, fast & responsive websites that grow your business'
  const ctaText = content?.cta_text || 'Get a Free Quote'
  const ctaLink = content?.cta_link || '#contact'

  useEffect(() => {
    const canvas = canvasRef.current
    if (!canvas) return

    const ctx = canvas.getContext('2d')
    if (!ctx) return

    canvas.width = window.innerWidth
    canvas.height = window.innerHeight

    const particles: Array<{x: number, y: number, vx: number, vy: number, size: number}> = []
    const particleCount = 80

    for (let i = 0; i < particleCount; i++) {
      particles.push({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        vx: (Math.random() - 0.5) * 0.5,
        vy: (Math.random() - 0.5) * 0.5,
        size: Math.random() * 2 + 1
      })
    }

    const primaryColor = settings?.primary_color || '#6C63FF'

    function animate() {
      ctx.clearRect(0, 0, canvas.width, canvas.height)
      
      particles.forEach((p, i) => {
        p.x += p.vx
        p.y += p.vy

        if (p.x < 0 || p.x > canvas.width) p.vx *= -1
        if (p.y < 0 || p.y > canvas.height) p.vy *= -1

        ctx.beginPath()
        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2)
        ctx.fillStyle = primaryColor + '40'
        ctx.fill()

        // Draw connections
        particles.forEach((p2, j) => {
          if (i === j) return
          const dx = p.x - p2.x
          const dy = p.y - p2.y
          const dist = Math.sqrt(dx * dx + dy * dy)
          
          if (dist < 120) {
            ctx.beginPath()
            ctx.moveTo(p.x, p.y)
            ctx.lineTo(p2.x, p2.y)
            ctx.strokeStyle = primaryColor + String(Math.floor((1 - dist / 120) * 20)).padStart(2, '0')
            ctx.stroke()
          }
        })
      })

      requestAnimationFrame(animate)
    }

    animate()

    const handleResize = () => {
      canvas.width = window.innerWidth
      canvas.height = window.innerHeight
    }

    window.addEventListener('resize', handleResize)
    return () => window.removeEventListener('resize', handleResize)
  }, [settings?.primary_color])

  return (
    <section className="relative min-h-screen flex items-center overflow-hidden animated-bg">
      {/* Canvas Background */}
      <canvas 
        ref={canvasRef}
        className="absolute inset-0 z-0"
      />

      {/* Gradient Orbs */}
      <div className="absolute top-1/4 left-1/4 w-72 h-72 bg-[#6C63FF]/20 rounded-full blur-[100px]" />
      <div className="absolute bottom-1/4 right-1/4 w-96 h-96 bg-[#00D9FF]/10 rounded-full blur-[120px]" />
      <div className="absolute top-1/2 right-1/3 w-48 h-48 bg-[#FF6B6B]/10 rounded-full blur-[80px]" />

      {/* Content */}
      <div className="relative z-10 max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-20">
        <div className="text-center max-w-4xl mx-auto">
          {/* Badge */}
          <div className="inline-flex items-center gap-2 bg-[#1C2333] border border-[#30363D] rounded-full px-4 py-1.5 mb-8 animate-fade-in">
            <Sparkles className="w-4 h-4 text-[#6C63FF]" />
            <span className="text-sm text-[#8B949E]">Trusted by 150+ businesses worldwide</span>
          </div>

          {/* Heading */}
          <h1 className="text-5xl md:text-7xl lg:text-8xl font-bold mb-6 leading-tight">
            <span className="gradient-text">{heading}</span>
          </h1>

          {/* Subheading */}
          <p className="text-lg md:text-xl text-[#8B949E] mb-10 max-w-2xl mx-auto">
            {subheading}
          </p>

          {/* CTA Buttons */}
          <div className="flex flex-col sm:flex-row items-center justify-center gap-4">
            <a 
              href={ctaLink}
              className="btn-primary text-lg px-8 py-4 flex items-center gap-2 group"
            >
              {ctaText}
              <ArrowRight className="w-5 h-5 group-hover:translate-x-1 transition-transform" />
            </a>
            <a 
              href="#services"
              className="btn-secondary text-lg px-8 py-4"
            >
              Our Services
            </a>
          </div>

          {/* Stats */}
          <div className="grid grid-cols-3 gap-8 mt-20 max-w-lg mx-auto">
            <div className="text-center">
              <div className="text-3xl font-bold gradient-text">200+</div>
              <div className="text-sm text-[#8B949E] mt-1">Projects</div>
            </div>
            <div className="text-center">
              <div className="text-3xl font-bold gradient-text">150+</div>
              <div className="text-sm text-[#8B949E] mt-1">Clients</div>
            </div>
            <div className="text-center">
              <div className="text-3xl font-bold gradient-text">99%</div>
              <div className="text-sm text-[#8B949E] mt-1">Satisfaction</div>
            </div>
          </div>
        </div>
      </div>

      {/* Floating Tech Icons */}
      <div className="absolute right-8 top-1/3 space-y-4 hidden lg:block">
        {[Zap, Shield, Sparkles].map((Icon, i) => (
          <div 
            key={i}
            className="w-12 h-12 rounded-xl bg-[#1C2333] border border-[#30363D] flex items-center justify-center animate-pulse"
            style={{ animationDelay: `${i * 0.3}s`, opacity: 0.5 }}
          >
            <Icon className="w-5 h-5 text-[#6C63FF]" />
          </div>
        ))}
      </div>
    </section>
  )
}
